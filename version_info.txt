# UTF-8
#
# Windows VersionInfo resource — embedded in workspace_agent.exe via
# PyInstaller's `version=` argument in the .spec. This is what Windows
# shows in:
#   - Properties → Details tab
#   - Firewall / Defender / SmartScreen prompts ("Do you want to allow
#     <FileDescription> to make changes / access the network?")
#   - Process Explorer / Task Manager (Description column)
#
# Without this resource, Windows just shows the bare filename
# ("workspace_agent.exe") in those prompts, which is confusing for end
# users. With it, the firewall popup says "Allow Workspace Agent to
# access the network" — recognizable.
#
# Format reference:
# https://learn.microsoft.com/en-us/windows/win32/menurc/versioninfo-resource
# https://pyinstaller.org/en/stable/usage.html#capturing-windows-version-data
#
VSVersionInfo(
  ffi=FixedFileInfo(
    # filevers / prodvers: 4-tuple (major, minor, patch, build). Keep
    # synced with `version` in pyproject.toml when bumping.
    filevers=(0, 1, 0, 0),
    prodvers=(0, 1, 0, 0),
    mask=0x3f,
    flags=0x0,
    OS=0x40004,         # VOS_NT_WINDOWS32
    fileType=0x1,       # VFT_APP
    subtype=0x0,
    date=(0, 0)
  ),
  kids=[
    StringFileInfo([
      StringTable(
        # 040904E4 = English (US) + Multilingual codepage.
        # Russian Windows still reads English version-info correctly;
        # using English keeps grep'ability + works across locales.
        '040904E4',
        [
          StringStruct('CompanyName',      'nnenoix'),
          StringStruct('FileDescription',  'Workspace Agent — локальный AI-агент для Google Workspace и российского e-commerce'),
          StringStruct('FileVersion',      '0.1.0'),
          StringStruct('InternalName',     'workspace_agent'),
          StringStruct('LegalCopyright',   'MIT License. Source: github.com/nnenoix/Assistant'),
          StringStruct('OriginalFilename', 'workspace_agent.exe'),
          StringStruct('ProductName',      'Workspace Agent'),
          StringStruct('ProductVersion',   '0.1.0'),
        ])
    ]),
    VarFileInfo([VarStruct('Translation', [1033, 1252])])
  ]
)
